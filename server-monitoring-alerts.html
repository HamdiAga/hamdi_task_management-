تنبيهات ذكية للمبرمج: كيف تراقب صحة سيرفرك وتتلقى إشعاراً فور تعطله

10 يونيو 2026 حمدي فواز أبو حاتم Automation & AI

أسوأ كابوس للمطور: أن تكتشف أن موقع عميلك معطل منذ 3 ساعات... عندما يتصل بك العميل غاضباً. المراقبة الذكية للسيرفر ليست رفاهية، بل ضرورة. في هذا الدليل، سأريك كيف تبني نظام مراقبة مجانياً، يفحص سيرفرك كل دقيقة، وينبهك فوراً على واتساب أو تليغرام إذا حدث أي خلل.

1. ماذا تراقب بالضبط؟

لا تكتفِ بـ "هل الموقع يعمل؟". راقب هذه المؤشرات الحيوية:

  • حالة HTTP: هل يرجع الموقع كود 200؟ أم 500؟ أم لا يستجيب أصلاً؟
  • استخدام CPU: إذا تجاوز 90% لأكثر من 5 دقائق، فهناك مشكلة.
  • استخدام RAM: اقتراب السيرفر من حد الذاكرة القصوى.
  • المساحة القرصية: امتلاء القرص يوقف قواعد البيانات فجأة.
  • حالة MySQL/Redis: هل الخدمات الأساسية تعمل؟
  • شهادة SSL: هل تقترب من الانتهاء؟

2. اسكريبت مراقبة شامل بلغة Bash

ضع هذا الاسكريبت على سيرفرك (أو سيرفر خارجي صغير) وشغّله كل دقيقة عبر cron job:

#!/bin/bash

# ========== الإعدادات ==========
SITE_URL="https://mysite.com"
TELEGRAM_BOT_TOKEN="your-bot-token"
TELEGRAM_CHAT_ID="your-chat-id"
CPU_THRESHOLD=90
RAM_THRESHOLD=90
DISK_THRESHOLD=90

# ========== دالة إرسال التنبيه ==========
send_alert() {
    local message="$1"
    curl -s -X POST "https://api.telegram.org/bot${TELEGRAM_BOT_TOKEN}/sendMessage" \
        -d chat_id="${TELEGRAM_CHAT_ID}" \
        -d text="🚨 تنبيه من السيرفر%0A${message}" \
        -d parse_mode="HTML" > /dev/null
}

# ========== 1. فحص HTTP ==========
HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" --max-time 10 "${SITE_URL}")
if [ "$HTTP_CODE" != "200" ]; then
    send_alert "الموقع لا يستجيب! كود HTTP: ${HTTP_CODE}"
fi

# ========== 2. فحص CPU ==========
CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1 | cut -d'.' -f1)
if [ "$CPU_USAGE" -gt "$CPU_THRESHOLD" ]; then
    send_alert "استخدام CPU مرتفع: ${CPU_USAGE}%"
fi

# ========== 3. فحص RAM ==========
RAM_USAGE=$(free | grep Mem | awk '{printf "%.0f", $3/$2 * 100}')
if [ "$RAM_USAGE" -gt "$RAM_THRESHOLD" ]; then
    send_alert "استخدام RAM مرتفع: ${RAM_USAGE}%"
fi

# ========== 4. فحص المساحة القرصية ==========
DISK_USAGE=$(df / | tail -1 | awk '{print $5}' | cut -d'%' -f1)
if [ "$DISK_USAGE" -gt "$DISK_THRESHOLD" ]; then
    send_alert "المساحة القرصية منخفضة: ${DISK_USAGE}%"
fi

# ========== 5. فحص MySQL ==========
if ! systemctl is-active --quiet mysql; then
    send_alert "خدمة MySQL متوقفة!"
fi

# ========== 6. فحص شهادة SSL (قبل 7 أيام من الانتهاء) ==========
SSL_EXPIRY=$(echo | openssl s_client -servername mysite.com -connect mysite.com:443 2>/dev/null | openssl x509 -noout -enddate | cut -d'=' -f2)
SSL_EXPIRY_EPOCH=$(date -d "${SSL_EXPIRY}" +%s)
NOW_EPOCH=$(date +%s)
DAYS_LEFT=$(( (SSL_EXPIRY_EPOCH - NOW_EPOCH) / 86400 ))
if [ "$DAYS_LEFT" -lt 7 ]; then
    send_alert "شهادة SSL ستنتهي خلال ${DAYS_LEFT} يوم!"
fi

3. نسخة Node.js للمطورين

إذا كنت تفضل JavaScript، فهذا اسكريبت Node.js يعمل بنفس المنطق:

const axios = require('axios');
const os = require('os');

const TELEGRAM_TOKEN = 'your-bot-token';
const CHAT_ID = 'your-chat-id';

async function sendAlert(message) {
    await axios.post(`https://api.telegram.org/bot${TELEGRAM_TOKEN}/sendMessage`, {
        chat_id: CHAT_ID,
        text: `🚨 ${message}`,
        parse_mode: 'HTML'
    });
}

async function checkSite() {
    try {
        const response = await axios.get('https://mysite.com', { timeout: 10000 });
        if (response.status !== 200) {
            await sendAlert(`الموقع يرجع كود HTTP: ${response.status}`);
        }
    } catch (error) {
        await sendAlert(`الموقع لا يستجيب: ${error.message}`);
    }
}

async function checkCPU() {
    const cpus = os.cpus();
    const totalIdle = cpus.reduce((acc, cpu) => acc + cpu.times.idle, 0);
    const totalTick = cpus.reduce((acc, cpu) => acc + cpu.times.user + cpu.times.nice + cpu.times.sys + cpu.times.irq + cpu.times.idle, 0);
    const idle = totalIdle / cpus.length;
    const total = totalTick / cpus.length;
    // هنا تحتاج لحساب الفرق بين قراءتين، هذا تبسيط
    const usage = 100 - (idle / total * 100);
    if (usage > 90) await sendAlert(`استخدام CPU مرتفع: ${usage.toFixed(1)}%`);
}

// شغّل كل الفحوصات
setInterval(async () => {
    await checkSite();
    await checkCPU();
    // أضف فحوصات RAM، Disk، إلخ...
    console.log('تم الفحص:', new Date().toISOString());
}, 60000); // كل دقيقة

4. أدوات مراقبة جاهزة (إذا لم ترد بناء نظامك)

  • UptimeRobot: مجاني لـ 50 مراقباً، يفحص موقعك كل 5 دقائق، ينبهك على الإيميل أو Slack أو تليغرام.
  • Netdata: أداة مراقبة شاملة تثبت على السيرفر خلال دقيقة واحدة، تعطيك لوحة تحكم حية بكل التفاصيل.
  • Laravel Telescope: لمشاريع Laravel، يريك كل استعلام، Job، Exception في الوقت الحقيقي.

قصة حقيقية: في أحد متاجر العملاء، امتلأ القرص الصلب بسبب ملفات logs ضخمة. توقفت MySQL فجأة، وتعطل الموقع بالكامل. اكتشفنا المشكلة بعد 4 ساعات عندما اشتكى العملاء. بعد تثبيت هذا الاسكريبت، صرنا نتلقى إشعاراً على تليغرام قبل أن تمتلأ المساحة، وننظف الـ logs فوراً. لم يتعطل الموقع بعدها أبداً.

خلاصة

راقب سيرفرك قبل أن يراقبك عملاؤك. اسكريبت Bash بسيط + cron job + تليغرام = نظام مراقبة مجاني وفعال. 15 دقيقة من الإعداد توفر عليك ساعات من الذعر وإصلاح الأعطال المفاجئة. الأفضل أن تعرف بالمشكلة قبل أن يعرفها زوار موقعك.
